# Backend servislerini başlat
cd ~/development/su-aritma/apps/mobile
flutter pub get

# İlk kurulum (sadece bir kez)
cd ~/development/su-aritma/apps/backend
cp env.example .env           
npm install
npx prisma migrate dev

# Backend API'yi çalıştır (default port: 4000, .env dosyasından değiştirilebilir)
cd ~/development/su-aritma/apps/backend
npm run dev

# Alternatif: Port 8080'de çalıştırmak için:
# PORT=8080 npm run dev

# Flutter uygulamasını çalıştır (ayrı terminalde)
cd ~/development/su-aritma/apps/mobile
flutter pub get

# Android Emülatör için:
cd ~/development/su-aritma/apps/mobile
flutter run --dart-define=API_BASE_URL=http://10.0.2.2:4000/api \
           --dart-define=ADMIN_ID=ALT-ADMIN-QA;


# Chrome/Web için:
cd ~/development/su-aritma/apps/mobile
flutter run -d chrome --dart-define=API_BASE_URL=http://localhost:4000/api \
           --dart-define=ADMIN_ID=ALT-ADMIN-QA

# iOS Simülatör için:
cd ~/development/su-aritma/apps/mobile
# Önce bağımlılıkları yükle:
flutter pub get
# CocoaPods bağımlılıklarını yükle (flutter pub get sonrası):
cd ios && pod install && cd ..
# Simülatörü aç (isteğe bağlı):
# open -a Simulator
# Mevcut cihazları kontrol et:
# flutter devices
# Device name ile çalıştır:
flutter run -d "iPhone 16e" --dart-define=API_BASE_URL=http://localhost:4000/api \  
           --dart-define=ADMIN_ID=ALT-ADMIN-QA
# Veya device ID ile:
# flutter run -d 9042F717-4746-4763-A15C-49A59588AF59 --dart-define=API_BASE_URL=http://localhost:4000/api --dart-define=ADMIN_ID=ALT-ADMIN-QA
